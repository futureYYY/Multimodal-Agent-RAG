1.安装了huggingface_cli库，可以使用进行安装。

pip install huggingface-cli

2.修改镜像源
# MacOS or Linux
export HF_ENDPOINT=https://hf-mirror.com
# Windows Powershell
$env:HF_ENDPOINT = "https://hf-mirror.com"

3.修改模型下载路径
export HF_HOME=/root/

3.huggingface-cli download --resume-download deepseek-ai/deepseek-vl-7b-chat (--local-dir ./models)
